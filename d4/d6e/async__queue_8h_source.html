77template<typename SEQNUM>
78class SeqNumDeJitter
79{
80#if defined(PLATFORM_DEBUG_BUILD)
81 rt::hash_set<SEQNUM> _Dedup;
82#endif
83 SEQNUM _Min;
84 SEQNUM _Max;
85 SIZE_T _GapCount;
86
87public:
88 // numbers starting from init + 1 are expected to be inserted
89 SeqNumDeJitter(SEQNUM init = 0){ Reset(init); }
90 void Reset(SEQNUM init)
91 { _GapCount = 0;
92 _Min = _Max = init;
93#if defined(PLATFORM_DEBUG_BUILD)
94 _Dedup.clear();
95#endif
96 }
97
105 bool Insert(SEQNUM s)
106 {
107 ASSERT(s > _Min && s != _Max); // s should have never been inserted before, i.e. not equal smaller than _Min or equals _Min or _Max, and
108#if defined(PLATFORM_DEBUG_BUILD)
109 ASSERT(_Dedup.find(s) == _Dedup.end()); // not in the inserted set between _Min and _Max
110#endif
111 if(s > _Max) // inserting a number beyond current min-max range
112 {
113 _GapCount += s - _Max - 1; // Adding the gap between _Max and s (excluding _Max and s)
114#if defined(PLATFORM_DEBUG_BUILD)
115 _Dedup.insert(_Max); // Now the range is expanded to _Min to s, _Max is in this range and was inserted before, so it should be in
116#endif
117 _Max = s; // expand _Max to s
118 }
119 else // inserting a number in min-max range
120 {
121 ASSERT(_GapCount);
122 _GapCount--;
123#if defined(PLATFORM_DEBUG_BUILD)
124 _Dedup.insert(s); // insert s
125#endif
126 }
127
128 if (_GapCount == 0) // This only happens if range between _Min and _Max already has no gap left and s == _Max + 1
129 {
130 _Min = _Max; // Set min-max range to s
131#if defined(PLATFORM_DEBUG_BUILD)
132 _Dedup.clear();
133#endif
134 return true;
135 }
136
137 ASSERT(_Max > _Min);
138 ASSERT(_Dedup.size() + _GapCount == _Max - _Min - 1); // this should always hold
139
140 return false;
141 }
142 SEQNUM DeJittered() const { return _Min; }
143 bool IsJittering() const { return _Min != _Max; }
144 SIZE_T GetMissingCount() const { return _GapCount; }
145 auto SeqNoMax() const { return _Min; }
146 auto SeqNoMin() const { return _Max; }
147};

293template<class T_QUEUE, UINT JITTER_BUFSIZE_MAX>
294class DeJitteredQueue // only for single reader
295{
296 static_assert((JITTER_BUFSIZE_MAX >= 4) && !(JITTER_BUFSIZE_MAX & (JITTER_BUFSIZE_MAX - 1)), "JITTER_BUFSIZE_MAX must be a power of 2 (and at least 4)");
297 typedef typename T_QUEUE::TYPE TYPE;
298protected:
299 static const UINT SEQ_RANGESIZE = 0x80000000U;
300 static const UINT SEQ_BITMASK = SEQ_RANGESIZE - 1U;
301 static const UINT FLAG_NOTNULL = 0x80000000U;
302
303 struct SeqT
304 { UINT SeqFlag;
305 TYPE Obj;
306 SeqT(){ SeqFlag = 0; }
307 SeqT(UINT seqflag, const TYPE& obj):Obj(obj){ SeqFlag = seqflag; ASSERT(IsNotNull()); }
308 UINT Seq() const { return SeqFlag&SEQ_BITMASK; }
309 bool IsNotNull() const { return SeqFlag&FLAG_NOTNULL; }
310 };
311
312 typename T_QUEUE::template WithType<SeqT> _Queue;
313
314 volatile UINT _NextSeq;
315
316 UINT _LastReadSeq;
317 rt::BufferEx<SeqT> _JitterBuf;
318 UINT _JitterBufBaseSeq;
319 static UINT _SeqMinus(UINT large, UINT small)
320 { ASSERT(large<SEQ_RANGESIZE && small<SEQ_RANGESIZE);
321 if(large >= small){ return large-small; }
322 else{ ASSERT(large < 0x40000000U && small > 0x40000000U);
323 return large + 0x80000000U - small;
324 } }
325 static UINT _SeqInc(UINT s){ return (s+1)&SEQ_BITMASK; }
326 UINT _WriterNextSeqFlag(){ return FLAG_NOTNULL|os::AtomicIncrement((volatile int*)&_NextSeq); }
327public:
328 DeJitteredQueue(){ _NextSeq = -1; _LastReadSeq = -1; _JitterBufBaseSeq = 0; }
329 template<typename ...ARGS>
330 auto Push(const TYPE& t, ARGS... args){ return _Queue.Push(SeqT(_WriterNextSeqFlag(),t), args...); }
331 template<typename ...ARGS>
332 bool Pop(TYPE& t, ARGS... args)
333 {
334 UINT expected_seq = _SeqInc(_LastReadSeq);
335 if(_JitterBuf.GetSize())
336 { // pop from jitter buffer if available
337 UINT idx = _SeqMinus(expected_seq, _JitterBufBaseSeq);
338 if(idx < _JitterBuf.GetSize() && _JitterBuf[idx].IsNotNull())
339 {
340 _LastReadSeq = expected_seq;
341 t = std::move(_JitterBuf[idx].Obj);
342 // shrink jitter buffer
343 if(idx + 1 == _JitterBuf.GetSize()){ _JitterBuf.ShrinkSize(0); }
344 else if(_JitterBuf.GetSize() > JITTER_BUFSIZE_MAX/2 && idx > _JitterBuf.GetSize()/2)
345 { _JitterBuf.erase(0, idx+1);
346 _JitterBufBaseSeq = (idx + 1 + _JitterBufBaseSeq)&SEQ_BITMASK;
347 }
348 return true;
349 }
350 }
351
352 SeqT pop;
353 while(_Queue.Pop(pop, args...))
354 {
355 if(pop.Seq() == expected_seq) // no jitter
356 {
357 _LastReadSeq = expected_seq;
358 t = std::move(pop.Obj);
359 return true;
360 }
361 // jittered
362 ASSERT(pop.IsNotNull());
363 UINT idx;
364 if(_JitterBuf.GetSize())
365 {
366 idx = _SeqMinus(pop.Seq(), _JitterBufBaseSeq);
367 ASSERT(idx < JITTER_BUFSIZE_MAX);
368 if(idx >= _JitterBuf.GetSize())VERIFY(_JitterBuf.ChangeSize(idx+1));
369 }
370 else
371 {
372 _JitterBufBaseSeq = expected_seq;
373 idx = _SeqMinus(pop.Seq(), _JitterBufBaseSeq);
374 ASSERT(idx < JITTER_BUFSIZE_MAX);
375 _JitterBuf.ChangeSize(idx+1);
376 }
377 _JitterBuf[idx] = pop;
378 }
379 return false;
380 }
