84template<int CRYTOMETHOD>
85class Keypair
86{
87public:
88 static const int KeySize = _details::ECC<CRYTOMETHOD>::KeySize;
89 typedef DataBlock<_details::ECC<CRYTOMETHOD>::KeySize, true> PrivateKey;
90 typedef DataBlock<_details::ECC<CRYTOMETHOD>::KeySize, false> PublicKey;
91
92protected:
93 PrivateKey _SK;
94 PublicKey _PK;
95
96public:
97 Keypair() = default;
98 Keypair(const rt::String_Ref& pk, const rt::String_Ref& sk){ _PK.FromBase32(pk); _SK.FromBase32(sk); }
99 const PublicKey& PK() const { return _PK; }
100 const PrivateKey& SK() const { return _SK; }
101 bool operator == (const Keypair& x){ return _SK == x._SK && _PK == x._PK; }
102 bool Generate(Botan::RandomNumberGenerator* pRNG = NULL)
103 { try{
104 rt::ObjectPlaceHolder<Botan::AutoSeeded_RNG> SysRNG;
105 if(pRNG == NULL){ pRNG = SysRNG.Init(); }
106
107 Botan::EC_Group group(_details::ECC<CRYTOMETHOD>::ECGroupName());
108 for(;;)
109 {
110 typename _details::ECC<CRYTOMETHOD>::PrivateKey sk(*pRNG, group);
111 if(sk.public_point().get_affine_x().get_bit(0) == sk.public_point().get_affine_y().get_bit(0))
112 {
113 memcpy(&_SK, sk.private_value().get_word_vector().data(), KeySize);
114 memcpy(&_PK, sk.public_point().get_affine_x().get_word_vector().data(), KeySize);
115 break;
116 }
117 }
118
119 return true;
120 }catch(...){ return false; }
121 }
122 bool SetPrivateKey(const PrivateKey& k)
123 { try{
124 Botan::AutoSeeded_RNG rng;
125 Botan::BigInt s;
126 Botan::BigIntAssign(s, k);
127 typename _details::ECC<CRYTOMETHOD>::PrivateKey sk(rng, Botan::EC_Group(_details::ECC<CRYTOMETHOD>::ECGroupName()), s);
128 memcpy(&_SK, sk.private_value().get_word_vector().data(), KeySize);
129 memcpy(&_PK, sk.public_point().get_affine_x().get_word_vector().data(), KeySize);
130 return true;
131 }catch(...){ return false; }
132 }
133 bool IsValid(bool deep_check = false) const
134 { try{
135 Botan::AutoSeeded_RNG rng;
136 Botan::BigInt s;
137 Botan::BigIntAssign(s, _SK);
138 typename _details::ECC<CRYTOMETHOD>::PrivateKey sk(rng, Botan::EC_Group(_details::ECC<CRYTOMETHOD>::ECGroupName()), s);
139 return sk.check_key(rng, deep_check);
140 }catch(...){ return false; }
141 }

235template<int CRYTOMETHOD>
236class _PublicKeyOp
237{
238protected:
239 rt::ObjectPlaceHolder<typename _details::ECC<CRYTOMETHOD>::PublicKey> _PK;
240public:
241 bool SetPublicKey(const typename Keypair<CRYTOMETHOD>::PublicKey& pk)
242 { try{
243 Botan::EC_Group group(_details::ECC<CRYTOMETHOD>::ECGroupName());
244 auto& curve = group.get_curve();
245
246 Botan::BigInt x, y;
247 Botan::BigIntAssign(x, pk);
248
249 //y = Botan::decompress_point(true, x, curve.get_curve());
250 Botan::BigInt xpow3 = x * x * x;
251 const Botan::BigInt& p = curve.get_p();
252
253 Botan::BigInt g = curve.get_a() * x;
254 g += xpow3;
255 g += curve.get_b();
256 g = g % p;
257
258 y = Botan::ressol(g, p);
259
260 if(y < 0)throw Botan::Illegal_Point("error during EC point decompression");
261 if(y.get_bit(0) != x.get_bit(0))y = p - y;
262
263 _PK.Reinit(group, Botan::PointGFp(curve, x, y));
264 return true;
265 }catch(...){}
266 return false;
267 }
268};
