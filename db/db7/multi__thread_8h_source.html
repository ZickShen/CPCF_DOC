278protected:
279 bool BeginUpdate(bool just_try = false)
280 {
281 if(just_try){ if(!_cs.TryLock())return false; }
282 else{ _cs.Lock(); }
283#if defined(PLATFORM_DEBUG_BUILD)
284 ASSERT(!_bModifying);
285 _bModifying = true;
286#endif
287 return true;
288 }
289 void EndUpdate(T* pNew = nullptr)/* nullptr indicates no change */
290 { if(pNew){ T* pOld = _p; _p = pNew; _SafeDel_Delayed(pOld, old_TTL); }
291#if defined(PLATFORM_DEBUG_BUILD)
292 _bModifying = false;
293#endif
294 _cs.Unlock();
295 }
296public:
297 typedef T t_Object;
298 ThreadSafeMutable()
299 { _p = nullptr;
300#if defined(PLATFORM_DEBUG_BUILD)
301 _bModifying = false;
302#endif
303 }
304 ~ThreadSafeMutable(){ Clear(); ASSERT(!_bModifying); }
305
306 const T& Get() const { static const T _t; return _p?*_p:_t; }
307 const T* operator -> () const { return &Get(); }
308 T* Clone() const { return _p?_New(T(*_p)):_New(T); }
309 T* New() const { return _New(T); }
310 void Clear(){ _cs.Lock(); _SafeDel_Delayed(_p, old_TTL); _cs.Unlock(); }
311 bool IsEmpty() const { return _p == nullptr; }
