197inline bool OnRecv(T* obj, rt::Buffer<BYTE>& buf)
198{
199 Datagram g;
200 g.RecvBuf = buf.Begin();
201 g.PeerAddressSize = sizeof(inet::InetAddrV6);
202 int len = 0;
203 do
204 {
205 len = ::recvfrom(obj->GetHandle(), g.RecvBuf, buf.GetSize(), 0, (sockaddr*)&g.PeerAddressFamily, &g.PeerAddressSize);
206 if(len>=0){ g.RecvSize = len; obj->OnRecv(&g);}
207 else { break; }
208 } while (true);
209 return len != 0 && (errno == EAGAIN || errno == EINTR);
210}
