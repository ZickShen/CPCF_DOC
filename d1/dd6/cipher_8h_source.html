153public:

156protected:

160 BYTE _Hash[NativeKeySize];
161 void _EnsureInit(CCCryptorRef& ccref, int op, int opt, LPCVOID vi = nullptr)
162 { if(ccref && opt == _CCRef_Opt)
163 { if(vi){ ASSERT(_CCRef_Opt == 0); CCCryptorReset(ccref, vi); }
164 return;
165 }
166 _CCRef_Opt = opt;
167 if(ccref)CCCryptorRelease(ccref);
168 CCCryptorCreate(op, kCCAlgorithmAES, _CCRef_Opt, _Hash, NativeKeySize, vi, &ccref);
169 }
170public:
171 Cipher(){ _CCRef_Encryption = _CCRef_Decryption = nullptr; }
172 ~Cipher(){ Empty(); }
173 void Empty()
174 { if(_CCRef_Encryption){ CCCryptorRelease(_CCRef_Encryption); _CCRef_Encryption = nullptr; }
175 if(_CCRef_Decryption){ CCCryptorRelease(_CCRef_Decryption); _CCRef_Decryption = nullptr; }
176 rt::Zero(_Hash);
177 }
178 static void ComputeKey(LPVOID key, LPCVOID data, UINT size){ Hash<_details::_AES_Traits<_METHOD>::KEY_HASHER>().Calculate(data, size, key); }
179 void SetKey(LPCVOID key, UINT len)
180 { Empty();
181 if(len != NativeKeySize){ ComputeKey(_Hash, key, len); }
182 else { memcpy(_Hash, key, len); }
183 }
184 void Encrypt(LPCVOID pPlain, LPVOID pCrypt, UINT Len)
185 { ASSERT((Len&_details::_AES_Traits<_METHOD>::DataAlign) == 0);
186 _EnsureInit(_CCRef_Encryption, kCCEncrypt, kCCOptionECBMode);
187 size_t out = 0;
188 CCCryptorUpdate(_CCRef_Encryption, pPlain, Len, pCrypt, Len, &out);
189 ASSERT(out == Len);
190 }

192 { ASSERT((Len&_details::_AES_Traits<_METHOD>::DataAlign) == 0);

196 ASSERT(out == Len);
197 }

199 { ASSERT((Len&_details::_AES_Traits<_METHOD>::DataAlign) == 0);

244class Cipher
245{
246 typename _details::_cipher_spec<_METHOD>::Cipher _Cipher;
247public:
248 static const UINT DataBlockSize = _details::_AES_Traits<_METHOD>::BlockSize;
249 static const UINT NativeKeySize = _details::_HashSize<_details::_AES_Traits<_METHOD>::KEY_HASHER>::size;
250 static void ComputeKey(LPVOID key, LPCVOID data, UINT size){ Hash<_details::_AES_Traits<_METHOD>::KEY_HASHER>().Calculate(data, size, key); }
251 void SetKey(LPCVOID key, UINT len)
252 { BYTE hash[NativeKeySize];
253 if(len != NativeKeySize){ ComputeKey(hash, key, len); key = hash; }
254 _Cipher.set_key((LPCBYTE)key, NativeKeySize);
255 if(len != NativeKeySize)rt::Zero(hash);
256 }
257 void Encrypt(LPCVOID pPlain, LPVOID pCrypt, UINT Len)
258 { ASSERT((Len%DataBlockSize) == 0);
259 _Cipher.encrypt_n((LPCBYTE)pPlain, (LPBYTE)pCrypt, Len/DataBlockSize);
260 }
261 void Decrypt(LPCVOID pCrypt, LPVOID pPlain, UINT Len)
262 { ASSERT((Len%DataBlockSize) == 0);
263 _Cipher.decrypt_n((LPCBYTE)pCrypt, (LPBYTE)pPlain, Len/DataBlockSize);
264 }
265 void EncryptBlockChained(LPCVOID pPlain, LPVOID pCrypt, UINT Len, UINT nonce)
266 { _details::CipherInitVec<DataBlockSize> IV(nonce);
267 ASSERT((Len%DataBlockSize) == 0);
268 auto* p = (const DataBlock<DataBlockSize>*)pPlain;
269 auto* c = (DataBlock<DataBlockSize>*)pCrypt;
270 Len /= DataBlockSize;
271 for(UINT i=0; i<Len; i++, p++, c++)
272 { IV ^= *p;
273 Encrypt(&IV, c, DataBlockSize);
274 c->CopyTo(IV);
275 }
276 }
277 void DecryptBlockChained(LPCVOID pCrypt, LPVOID pPlain, UINT Len, UINT nonce)
278 { _details::CipherInitVec<DataBlockSize> IV(nonce);
279 ASSERT((Len%DataBlockSize) == 0);
280 const DataBlock<DataBlockSize>* iv = &IV;
281 auto* p = (DataBlock<DataBlockSize>*)pPlain;
282 auto* c = (const DataBlock<DataBlockSize>*)pCrypt;
283 Len /= DataBlockSize;
284 for(UINT i=0; i<Len; i++, p++, c++)
285 { Decrypt(c, p, DataBlockSize);
286 *p ^= *iv;
287 iv = c;
288 }
289 }
