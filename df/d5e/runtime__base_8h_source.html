393INLFUNC int PopCount(ULONGLONG x) {
394 x -= (x >> 1) & _details::m1;
395 x = (x & _details::m2) + ((x >> 2) & _details::m2);
396 x = (x + (x >> 4)) & _details::m4;
397 x += x >> 8;
398 x += x >> 16;
399 x += x >> 32;
400 return x & 0x7f;
401}

422 INLFUNC unsigned int xxHash_small(const void* key, int len, unsigned int seed)
423 { const unsigned char* p = (unsigned char*)key;
424 const unsigned char* const bEnd = p + len;
425 unsigned int idx = seed + PRIME1;
426 unsigned int crc = PRIME5;
427 const unsigned char* const limit = bEnd - 4;
428 while (p<limit)
429 { crc += ((*(unsigned int*)p) + idx++);
430 crc += _rotl(crc, 17) * PRIME4;
431 crc *= PRIME1;
432 p+=4;
433 }
434 while (p<bEnd)
435 { crc += ((*p) + idx++);
436 crc *= PRIME1;
437 p++;
438 }
439 crc += len;
440 crc ^= crc >> 15;
441 crc *= PRIME2;
442 crc ^= crc >> 13;
443 crc *= PRIME3;
444 crc ^= crc >> 16;
445 return crc;
446 }

449INLFUNC UINT xxHash(const void* input, int len, unsigned int seed = 0)
450{ // Special case, for small inputs
451 if (len < 16) return _details::xxHash_small(input, len, seed);
452 const unsigned char* p = (const unsigned char*)input;
453 const unsigned char* const bEnd = p + len;
454 unsigned int v1 = seed + _details::PRIME1;
455 unsigned int v2 = v1 * _details::PRIME2 + len;
456 unsigned int v3 = v2 * _details::PRIME3;
457 unsigned int v4 = v3 * _details::PRIME4;
458 const unsigned char* const limit = bEnd - 16;
459 unsigned int crc;
460 while (p<limit)
461 { v1 = _rotl(v1, 13) + (*(unsigned int*)p); p+=4;
462 v2 = _rotl(v2, 11) + (*(unsigned int*)p); p+=4;
463 v3 = _rotl(v3, 17) + (*(unsigned int*)p); p+=4;
464 v4 = _rotl(v4, 19) + (*(unsigned int*)p); p+=4;
465 }
466
467 p = bEnd - 16;
468 v1 += _rotl(v1, 17); v2 += _rotl(v2, 19); v3 += _rotl(v3, 13); v4 += _rotl(v4, 11);
469 v1 *= _details::PRIME1; v2 *= _details::PRIME1; v3 *= _details::PRIME1; v4 *= _details::PRIME1;
470 v1 += *(unsigned int*)p; p+=4; v2 += *(unsigned int*)p; p+=4; v3 += *(unsigned int*)p; p+=4; v4 += *(unsigned int*)p; // p+=4;
471 v1 *= _details::PRIME2; v2 *= _details::PRIME2; v3 *= _details::PRIME2; v4 *= _details::PRIME2;
472 v1 += _rotl(v1, 11); v2 += _rotl(v2, 17); v3 += _rotl(v3, 19); v4 += _rotl(v4, 13);
473 v1 *= _details::PRIME3; v2 *= _details::PRIME3; v3 *= _details::PRIME3; v4 *= _details::PRIME3;
474
475 crc = v1 + _rotl(v2, 3) + _rotl(v3, 6) + _rotl(v4, 9);
476 crc ^= crc >> 11;
477 crc += (_details::PRIME4+len) * _details::PRIME1;
478 crc ^= crc >> 15;
479 crc *= _details::PRIME2;
480 crc ^= crc >> 13;
481
482 return crc;
483}
