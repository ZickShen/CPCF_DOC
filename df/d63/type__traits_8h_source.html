551struct Bits
552{ static bool IsZero(LPCBYTE p)
553 { return 0 == *(size_t*)p && Bits<len - sizeof(size_t)>::IsZero(p + sizeof(size_t));
554 }
555 static bool IsVoid(LPCBYTE p)
556 { return ((size_t)-1) == *(size_t*)p && Bits<len - sizeof(size_t)>::IsVoid(p + sizeof(size_t));
557 }
