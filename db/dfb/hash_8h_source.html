182template<int HASH_METHOD = HASH_MD5>
183class Hash
184{
185public:
186 static const int HASHSIZE = _details::_HashSize<HASH_METHOD>::size;
187 typedef typename _details::_HashTrait<HASH_METHOD>::type HashType;
188protected:
189 HashType Hasher;
190public:
191 auto& Reset()
192 { try{ Hasher.clear(); }
193 catch(std::exception& e){ _LOG_EXPCEPTION(e.what()); }
194 return *this;
195 }
196 auto& Update(LPCVOID data, SIZE_T size)
197 { try{ Hasher.update((LPCBYTE)data, size); }
198 catch(std::exception& e){ _LOG_EXPCEPTION(e.what()); }
199 return *this;
200 }
201 auto& Finalize(LPVOID HashValue)
202 { try{ Hasher.final((LPBYTE)HashValue); }
203 catch(std::exception& e){ _LOG_EXPCEPTION(e.what()); }
204 return *this;
205 }
206 auto& Calculate(LPCVOID data, SIZE_T size, LPVOID HashValue)
207 { try
208 { Hasher.clear();
209 Hasher.update((LPCBYTE)data, size);
210 Hasher.final((LPBYTE)HashValue);
211 }
212 catch(std::exception& e){ _LOG_EXPCEPTION(e.what()); }
213 return *this;
214 }
215 auto& Calculate(LPCVOID data, SIZE_T size, rt::String& hash_hex)
216 {
217 BYTE hash[HASHSIZE];
218 Calculate(data,size,hash);
219
220 hash_hex.SetLength(HASHSIZE*2);
221 os::Base16Encode(hash_hex.Begin(), hash, HASHSIZE);
222 return *this;
223 }
224 template<typename T> auto& Update(const T& x){ STATIC_ASSERT(rt::TypeTraits<T>::IsPOD); Update(&x, sizeof(x)); return *this; }
225};
